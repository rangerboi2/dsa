1)Write a java program to perform following string operations: 
  i. Length  ii. Copy iii. Concatenation iv. Compare v. Reverse

import java.util.Scanner;

public class MenuDrivenStringOperations {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String str1, str2;
        int choice;

        do {
            System.out.println("Menu:");
            System.out.println("1. Length of a String");
            System.out.println("2. Copy a String");
            System.out.println("3. Concatenate Strings");
            System.out.println("4. Compare Strings");
            System.out.println("5. Reverse a String");
            System.out.println("6. Exit");
            System.out.print("Enter your choice: ");
            choice = scanner.nextInt();

            switch (choice) {
                case 1:
                    System.out.print("Enter a string: ");
                    str1 = scanner.next();
                    System.out.println("Length of the string: " + str1.length());
                    break;
                case 2:
                    System.out.print("Enter a string: ");
                    str1 = scanner.next();
                    str2 = str1;
                    System.out.println("Copied string: " + str2);
                    break;
                case 3:
                    System.out.print("Enter first string: ");
                    str1 = scanner.next();
                    System.out.print("Enter second string: ");
                    str2 = scanner.next();
                    System.out.println("Concatenated string: " + str1.concat(str2));
                    break;
                case 4:
                    System.out.print("Enter first string: ");
                    str1 = scanner.next();
                    System.out.print("Enter second string: ");
                    str2 = scanner.next();
                    int compareResult = str1.compareTo(str2);
                    if (compareResult == 0)
                        System.out.println("Strings are equal.");
                    else
                        System.out.println("Strings are not equal.");
                    break;
                case 5:
                    System.out.print("Enter a string: ");
                    str1 = scanner.next();
                    StringBuilder reversedStr = new StringBuilder(str1);
                    reversedStr.reverse();
                    System.out.println("Reversed string: " + reversedStr);
                    break;
                case 6:
                    System.out.println("Exiting program...");
                    break;
                default:
                    System.out.println("Invalid choice! Please enter a valid option.");
            }
        } while (choice != 6);

        scanner.close();
    }
}
